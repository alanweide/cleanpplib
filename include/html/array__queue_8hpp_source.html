29  void resize_if_needed() {
30  int old_cap = cap_;
31  if (length_ >= cap_) {
32  cap_ = length_ * GROW_FAC;
33  } else if (length_ * SHRINK_FAC < cap_) {
34  cap_ /= SHRINK_FAC;
35  if (cap_ < MIN_CAP) {
36  cap_ = MIN_CAP;
37  }
38  }
39  if (cap_ != old_cap) {
40  std::unique_ptr<T[]> temp_cont = std::move(contents_);
41  contents_ = std::make_unique<T[]>(cap_);
42  for (int i = 0; i < length_; i++) {
43  contents_[i] = std::move(temp_cont[(i + head_) % old_cap]);
44  }
45  }
46  }
