18 template <class T>
19 class list_kernel_impl: public clean_base {
20  /*
21  list_kernel is modeled by (prec: string of T,
22  rem: string of T)
23  */
24 public:
25 
26  /*
27  updates this
28  requires |this.rem| > 0
29  ensures exists A, B, x : #this = (A * <x>, B) : this = (A, <x> * B)
30  */
31  virtual void advance() = 0;
32 
33  /*
34  updates this
35  requires |this.prec| > 0
36  ensures exists A, B, x : #this = (A, <x> * B) : this = (A * <x>, B)
37  */
38  virtual void retreat() = 0;
39 
40  /*
41  updates this
42  clears x
43  ensures this = (#this.prec * <x>, #this.rem)
44  */
45  virtual void insert(T&& x) = 0;
46 
47  /*
48  updates this
49  ensures #this = (this.prec * <remove>, this.rem)
50  */
51  virtual T remove() = 0;
52 
53  /*
54  ensures is_at_end = (|this.rem| = 0)
55  */
56  virtual bool is_at_end() const = 0;
57 
58  /*
59  ensures is_at_front = (|this.prec| = 0)
60  */
61  virtual bool is_at_front() const = 0;
62 
63  virtual std::string to_str() {
64  std::stringstream out;
65 
66  // Reset this and count position
67  int pos = 0;
68  while (!this->is_at_front()) {
69  this->retreat();
70  pos++;
71  }
72 
73  // Append first part of this to out
74  out << "(<";
75  while (!this->is_at_end() && pos > 0) {
76  T x;
77  x = this->remove();
78  out << x;
79  this->insert(std::move(x));
80  pos--;
81  if (pos != 0) {
82  out << ", ";
83  }
84  }
85  out << ">, ";
86 
87  // Append second part of this to out
88  out << "<";
89  while (!this->is_at_end()) {
90  T x;
91  x = this->remove();
92  out << x;
93  this->insert(std::move(x));
94  pos++;
95  if (!this->is_at_end()) {
96  out << ", ";
97  }
98  }
99  out << ">)";
100 
101  // Reset this this to original position
102  while (pos > 0) {
103  this->retreat();
104  pos--;
105  }
106 
107  return out.str();
108  }
109 };
